12 /**
13  * CLIFormatter is a utility object that provides reusable functions
14  * for rendering stylized CLI elements such as boxes, menus, and ASCII titles.
15  * It helps enforce visual consistency and reduces duplication in CLI views.

19   /**
20    * Prints a boxed section to the console with a title and multiline body.
21    *
22    * @param title the title to be displayed at the top of the box
23    * @param body  the list of lines to be printed inside the box
24    */
25   def printBoxedContent(title: String, body: List[String]): Unit  =
26     val contentWidth = (title.length :: body.map(_.length)).max + 4
27     println("╭" + "─" * contentWidth + "╮")
28     println(s"│  $title" + " " * (contentWidth - title.length - 2) + "│")
29     println("├" + "─" * contentWidth + "┤")
30     body.foreach { line =>
31       println(s"│  $line" + " " * (contentWidth - line.length - 2) + "│")
32     }
33     println("╰" + "─" * contentWidth + "╯")

35   /**
36    * Prints a menu list with indexed options inside a styled box,
37    * and prompts the user to select an option.
38    *
39    * @param title the title displayed at the top of the menu
40    * @param body  the list of menu options to display
41    */
42   def printBoxedMenu(title: String, body: List[String]): Unit  =
43     val width = (title.length :: body.map(_.length + 6)).max + 4
44     println("╭" + "─" * width + "╮")
45     println(s"│  $title" + " " * (width - title.length - 2) + "│")
46     println("├" + "─" * width + "┤")
47     body.zipWithIndex.foreach { case (option, i) =>
48       val line = f"$i. $option"
49       println(s"│  $line" + " " * (width - line.length - 2) + "│")
50     }
51     println("╰" + "─" * width + "╯")
52     print("Insert your choice > ")

63   /**
64    * Prints a grid of String from a map (WorldMap)
65    *
66    * @param map worldMap of a cities
67    * @param conquered string also contained in map, to change color
68    */
69   def printMap(map: WorldMap, conquered: Set[String]): Unit =
70     val mapString = (0 until map.getSizeOfTheMap).map { y =>
71       (0 until map.getSizeOfTheMap).map { x =>
72         map.findInMap { case (_, coords) => coords.contains((x, y)) } match
73           case Some(city) =>
74             val name = city
75             if conquered.contains(city) then s"❗"
76             else name
77           case None => "/"
78       }.mkString(" ")
79     }.mkString("\n    ")
80     println("    " + mapString)
