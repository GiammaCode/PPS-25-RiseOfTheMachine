82   /**
83    * Applies the given action to the player and the world map, producing a result.
84    */
85   override def executeAction(action: ValidAction, worldMap: WorldMap): ExecuteActionResult[Self] =
86     action match
87     case CityDefense(targets) =>
88       val updated = withDefendedCities(targets.toSet).addAction(action)
89       val defendedCities: List[City] = targets.flatMap(cityName =>
90         worldMap.getCityByName(cityName).map(_.defenseCity(CityDefenseBoost))
91       )
92       result(updated, Some(defendedCities), s"CityDefense on: ${targets.mkString(", ")}")
93 
94     case GlobalDefense(targets) =>
95       val updated = withDefendedCities(defendedCities).addAction(action)
96       val updatedCities = targets.flatMap(worldMap.getCityByName).map(_.defenseCity(GlobalDefenseBoost))
97       result(updated, Some(updatedCities), "GlobalDefense executed")
98 
99     case DevelopKillSwitch =>
100       val updated = copy(killSwitch = killSwitch + KillSwitchIncrement).addAction(action)
101       result(updated, None, "KillSwitch progress increased")
102
