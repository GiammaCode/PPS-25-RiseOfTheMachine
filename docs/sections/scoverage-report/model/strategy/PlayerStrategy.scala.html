8 /**
9  * Represents the relative weights (probabilities) assigned to each possible HumanAction
10  * that a strategy can choose from during decision-making.
11  *
12  * These weights are used to influence the likelihood of each action being selected
13  * by weighted random selection logic.
14  *
15  * The higher the weight, the more likely the action is to be chosen.
16  * A weight of zero effectively disables that action.
17  *
18  * @param cityDefenseWeight     the weight assigned to the CityDefense action
19  * @param globalDefenseWeight   the weight assigned to the GlobalDefense action
20  * @param killSwitchWeight      the weight assigned to the DevelopKillSwitch action
21  */
22 case class ActionProbabilities(
23                                 cityDefenseWeight: Int,
24                                 globalDefenseWeight: Int,
25                                 killSwitchWeight: Int
26                               )

44 /**
45  * A smart strategy implementation for the Human player.
46  * The decision is based on the difficulty level defined in the game settings.
47  *
48  * - Easy: defensive/passive
49  * - Normal: random among all valid actions
50  * - Hard: risk-based and proactive
51  */
52 object SmartHumanStrategy extends PlayerStrategy[HumanAction]:
53   /**
54    * Chooses the appropriate decision strategy based on the difficulty level.
55    *
56    * @param state the current world state
57    * @return the chosen HumanAction
58    */
59   override def decideAction(state: WorldState)(using probs: ActionProbabilities): HumanAction =
60     def decideWeighted(targetSelector: WorldState => List[String]): HumanAction =
61       val targets = targetSelector(state)
62       val defend = nonDefended(targets, state).take(1)
63       val available = baseHumanActions(defend, targets)
64       val actions = available.collect {
65         case a: CityDefense => (a, probs.cityDefenseWeight)
66         case a: GlobalDefense => (a, probs.globalDefenseWeight)
67         case DevelopKillSwitch => (DevelopKillSwitch, probs.killSwitchWeight)
68       }
69       chooseWeighted(actions)
70 
71     state.difficulty match
72       case Difficulty.Easy => decideWeighted(possibleTargets)
73       case Difficulty.Normal => decideWeighted(possibleTargets)
74       case Difficulty.Hard => decideWeighted(topRiskTargets)

107   /**
108    * Builds the list of available actions for the human player,
109    * including city defense, kill switch development, and global defense.
110    *
111    * @param defend     list of cities to defend
112    * @param allTargets list of all potential city targets
113    * @return a list of valid HumanActions
114    */
115   private def baseHumanActions(defend: List[String], allTargets: List[String]): List[HumanAction] =
116     List(
117       Option.when(defend.nonEmpty)(CityDefense(defend)),
118       Some(DevelopKillSwitch),
119       Some(GlobalDefense(allTargets))
120     ).flatten

122   /**
123    * Selects one action from a list of weighted HumanActions using a weighted random algorithm.
124    *
125    * Each action in the input list is associated with a weight (priority or likelihood).
126    * The selection is made such that the probability of choosing an action is proportional to its weight.
127    *
128    * @param actions a list of tuples where each tuple contains a HumanAction and its corresponding weight
129    * @return one HumanAction randomly selected based on the specified weights
130    */
131   private def chooseWeighted(actions: List[(HumanAction, Int)]): HumanAction =
132     val total = actions.map(_._2).sum
133     val rand = Random.nextInt(total)
134     actions
135       .scanLeft((Option.empty[HumanAction], 0)) {
136         case ((_, acc), (action, weight)) => (Some(action), acc + weight)
137       }
138       .collectFirst { case (Some(action), acc) if rand < acc => action }
139       .get
