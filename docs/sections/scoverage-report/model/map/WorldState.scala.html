101     /**
102      * Checks whether the game is over.
103      * The game ends if more than 10 cities are infected.
104      *
105      * @return true if game over, false otherwise and who wins the game
106      */
107     def isGameOver: (Boolean, Option[PlayerEntity]) = ws match
108       case State(map, ai, human, _, _) =>
109         val totalCities = map.numberOfCity()
110         val conqueredPercentage = (map.numberOfCityInfected().toDouble / totalCities) * 100
111         val killSwitchProgress = human.killSwitch
112         val capitalConquered = map.capitalConqueredCounter
113 
114         if conqueredPercentage >= VictoryThresholdPercent || capitalConquered == MaxCapitalsConquered then (true, Some(ai))
115         else if killSwitchProgress >= KillSwitchVictoryThreshold then (true, Some(human))
116         else (false, None)

206     /**
207      * Returns the success percentage for a given city and action type.
208      *
209      * @param cityName   the name of the target city
210      * @param action the action type: "infect" or "sabotage" or "evolve"

213     def probabilityByCityandAction(cityName: String, action: AiAction): Int =
214       attackableCities.find(_._1 == cityName).map{
215         case(_,infectChance, sabotageChance) => action match

217           case Sabotage(_) => sabotageChance

221     /**
222      * Private method to calculate a percentage of success
223      * of an attack.
224      *
225      * @param cityDefense       the value of city defense
226      * @param playerAttackValue the value of player attack
227      * @return a value to success
228      */
229     private def calculatePercentageOfSuccess(cityDefense: Int, playerAttackValue: Int, difficulty: Difficulty): Int =
230       val rawValue = difficulty match
231       case Difficulty.Easy => EasyBaseSuccess - cityDefense + playerAttackValue
232       case Difficulty.Normal => NormalBaseSuccess - cityDefense + playerAttackValue
233       case Difficulty.Hard => HardBaseSuccess - cityDefense + playerAttackValue
234 
235       Math.max(MinSuccessPercent, Math.min(MaxSuccessPercent, rawValue))
