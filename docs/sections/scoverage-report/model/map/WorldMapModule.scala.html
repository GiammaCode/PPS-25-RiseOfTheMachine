10 /**
11  * == WorldMapModule ==
12  *
13  * This module defines the representation and construction logic for the game map.
14  * A world map consists of cities, each associated with a set of coordinate tiles representing its spatial footprint.
15  *
16  * === Opaque Type ===
17  * - [[WorldMap]] is an abstracted representation of a set of cities with their tile coordinates.
18  *
19  *
20  * == Traits ==
21  * === CreateModuleType ===
22  * Abstract interface for any map creation strategy.
23  *
24  * == DeterministicMapModule ==
25  * Creates a reproducible, grid-based map with adjacent tile allocation for each city.
26  *
27  * == UndeterministicMapModule ==
28  * Uses a state monad with a PRNG to construct a randomized map, including random tile distribution and city sizes.
29  *
30  * == Extension Methods ==
31  * Adds useful methods to `WorldMap`:
32  *  - `getSizeOfTheMap`: returns map dimensions
33  *  - `getCityByName(name)`: looks up a city by name
34  *  - `numberOfCityInfected`: counts infected (AI-controlled) cities
35  *  - `getAdjacentCities`: finds human cities adjacent to infected ones
36  *  - `changeACityOfTheMap`: replaces a city with an updated version
37  */

79        * Expands a city from a starting coordinate up to a desired size,
80        * ensuring all cells are connected.
81        */
82       def expandCity(start: Coord, available: Set[Coord], desiredSize: Int): Set[Coord] =
83         @tailrec
84         def expandLoop(frontier: List[Coord], visited: Set[Coord]): Set[Coord] =
85           if visited.size >= desiredSize || frontier.isEmpty then visited
86           else
87             val next = frontier.head
88             val newNeighbors = adjacentTo(Set(next), size).toSet.intersect(available).diff(visited)
89             expandLoop(frontier.tail ++ newNeighbors, visited + next)
90 
91         expandLoop(List(start), Set(start)).intersect(available)

93       /**
94        * Recursively builds the map by placing cities one by one.
95        */
96       @tailrec
97       def createMapLoop(index: Int, available: Set[Coord], acc: WorldMap): WorldMap =
98         val maxCellPossible = 10
99         if available.isEmpty then acc
100         else
101           val isCapital = index < nCapital
102           val name = letterAt(index, isCapital)
103           val city = createCity(name, size, isCapital)
104           val start = available.head
105           val maxCitySize = math.min(maxCellPossible, available.size)
106           val tiles = expandCity(start, available, maxCitySize)
107           if tiles.size < 0 then createMapLoop(index + 1, available -- tiles, acc)
108           else createMapLoop(index + 1, available -- tiles, acc + (city -> tiles))
109 
110       createMapLoop(0, getSetOfCoords(size), Set.empty)

128

159     /**
160      * Grows a city from a start tile to a maximum size by exploring adjacent tiles.
161      */
162     private def growCity(start: Coord,
163                          maxSize: Int,
164                          available: Set[Coord],
165                          mapSize: Int
166                         ): Set[Coord] =
167       @tailrec
168       def expand(frontier: List[Coord], built: Set[Coord]): Set[Coord] =
169         if built.size >= maxSize || frontier.isEmpty then built
170         else
171           val next = frontier.head
172           val neighbors = adjacentTo(built, mapSize).filter(c =>
173             c._1 >= 0 && c._2 >= 0 &&
174               c._1 < mapSize && c._2 < mapSize &&
175               available.contains(c) && !built.contains(c)
176           )
177           expand(frontier.tail ++ neighbors, built + next)
178 
179       expand(List(start), Set(start))

182      * Generates the a lazy list of map and returns the first valid one that fills the map grid and respect the parameter.
183      */
184     override def createMap(size: Int): WorldMap =
185       val minNumberOfCity = 10
186       val minCapitalSize = 3
187       val maxAttempts = 1000
188       val maps = placeCities(size, getSetOfCoords(size)).take(maxAttempts)
189       maps.find(m =>

191         m.count( c => c._1.isCapital) == nCapital  &&
192         m.forall(c => !c._1.isCapital || c._2.size >= minCapitalSize))
193         .orElse(

206    * Returns valid neighboring coordinates that are not blocked, based on a seed set.

208   private def validNeighbors(seeds: List[Coord], size: Int, blocked: Set[Coord]): List[Coord] =
209     for
210       (x, y) <- seeds
211       dx <- -1 to 1
212       dy <- -1 to 1
213       if dx != 0 || dy != 0
214       nx = x + dx
215       ny = y + dy
216       if 0 <= nx && nx < size
217       if 0 <= ny && ny < size
218       if !blocked.contains((nx, ny))
