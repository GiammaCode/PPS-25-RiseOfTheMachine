9   /**
10    * A typeclass representing Monad behavior for any type constructor `M[_]`.
11    * A monad must define `unit` and `flatMap`, and optionally `map`.
12    */
13   trait Monad[M[_]]:
14     /**
15      * Lifts a value into the monadic context.
16      *
17      * @param a the value to lift
18      * @return a monadic value containing `a`
19      */
20     def unit[A](a: A): M[A]
21 
22     /**
23      * Extension methods available on any monadic value `M[A]`.
24      */
25     extension [A](m: M[A])
26       /**
27        * Chains a computation that returns a monadic value.
28        *
29        * @param f a function from A to M[B]
30        * @return the result of applying `f` and flattening
31        */
32       def flatMap[B](f: A => M[B]): M[B]
33 
34       /**
35        * Transforms the value inside the monad.
36        *
37        * @param f a function from A to B
38        * @return a monad containing the transformed value
39        */
40       def map[B](f: A => B): M[B] = m.flatMap(a => unit(f(a)))

43     /**
44      * Combines two monadic values with a binary function.
45      *
46      * @param m first monadic value
47      * @param m2 second monadic value
48      * @param f function to combine values from `m` and `m2`
49      * @return combined monadic result
50      */
51     private def map2[M[_]: Monad, A, B, C](m: M[A], m2: M[B])(f: (A, B) => C): M[C] =
52       m.flatMap(a => m2.map(b => f(a, b)))
53 
54     /**
55      * Sequences two monadic computations, discarding the result of the first.
56      *
57      * @param m first monadic value
58      * @param m2 second monadic computation (lazy)
59      * @return the result of the second computation
60      */
61     private def seq[M[_]: Monad, A, B](m: M[A], m2: => M[B]): M[B] =
62       map2(m, m2)((_, b) => b)

75     case class State[S, A](run: S => (S, A))

77     /**

82        * Runs the stateful computation with an initial state.
83        *
84        * @param s the initial state

86        */
87       def apply(s: S): (S, A) = m match
88         case State(run) => run(s)

90       /**

96        */
97       def withFilter(p: A => Boolean): State[S, A] =
98         State { s =>

100           if p(a) then (s2, a)

102         }
