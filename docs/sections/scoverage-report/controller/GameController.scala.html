63   /**
64    * Executes an AI action, updating the state if the probability check passes.
65    *
66    * @param action the AI action to execute
67    * @param prob the success probability of the action
68    * @return a state transformation
69    */
70   private[controller] def doPlayerAction(action: AiAction, prob: Int): State[GameState, Unit] =
71     import model.util.Util.doesTheActionGoesRight
72     if doesTheActionGoesRight(prob)
73     then State(gs =>
74       val currentWorldState = gs.worldState
75       val actionResult = currentWorldState.playerAI.executeAction(action, currentWorldState.worldMap)
76       val updatedWorldState = currentWorldState
77         .updatePlayer(actionResult.getPlayer)
78         .updateMap(actionResult.getCities)
79       (gs.copy(worldState = updatedWorldState), ())
80     )
81     else State(gs => (gs, ()))

83   /**
84    * Executes the human player's action. If none is provided, a default strategy is used.
85    *
86    * @param maybeAction an optional explicit action; if none, a strategy will decide
87    * @return a state transformation
88    */
89   private[controller] def doHumanAction(maybeAction: Option[HumanAction]): State[GameState, Unit] =
90     State(gs =>
91       val currentWorldState = gs.worldState
92       val action = maybeAction.getOrElse(currentWorldState.playerHuman.decideActionByStrategy(currentWorldState))
93       val actionResult = currentWorldState.playerHuman.executeAction(action, currentWorldState.worldMap)
94       val updatedState = currentWorldState
95         .updateHuman(actionResult.getPlayer)
96         .updateMap(actionResult.getCities)
97       (gs.copy(worldState = updatedState), ())
98     )

110   /**
111    * Renders a turn and processes user input to determine actions.
112    *
113    * @return a [[TurnResult]] containing the resolved actions for the turn
114    */
115   private def renderTurn(using GameSettings): State[GameState, TurnResult] = State { gs =>
116     val currentWorldState = gs.worldState.updateTurn
117     import view.ViewModule.CLIView
118     val turnInput = CLIView.renderGameTurn(currentWorldState)
119 
120     def resolveAction[A <: TurnAction](index: Int, city: String, options: List[A])(using InputHandler.ActionResolver[A]) =
121       InputHandler.getActionFromChoice(index, CityContext(city, currentWorldState.attackableCities.map(_._1)), options)
122 
123     val playerResult = resolveAction(turnInput.aiInput_action, turnInput.aiInput_city, currentWorldState.playerAI.getPossibleAction)
124     val humanResultOpt = turnInput.humanInput.map(resolveAction(_, _, currentWorldState.playerHuman.getPossibleAction))
125 
126     (playerResult, humanResultOpt) match
127       case (Right(playerAction), humanOpt) if humanOpt.forall(_.isRight) =>
128         val probability = currentWorldState.probabilityByCityandAction(turnInput.aiInput_city, playerAction)
129         val humanActionOpt = humanOpt.flatMap(_.toOption)
130         (gs.copy(worldState = currentWorldState), TurnResult(playerAction, probability, humanActionOpt))
131       case _ =>
132         println("Invalid input. Retrying turn.")
133         renderTurn.run(gs)
134   }

136   /**
137    * Executes a full game turn by processing the AI and human actions.
138    *
139    * @return a transformation that updates the game state
140    */
141   def gameTurn(using GameSettings): State[GameState, Unit] =
142     for
143       turn <- renderTurn
144       _ <- doPlayerAction(turn.playerAction, turn.playerProb)
145       _ <- doHumanAction(turn.humanAction)
146     yield ()
