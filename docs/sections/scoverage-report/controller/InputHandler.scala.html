29   /** Typeclass for resolving a base action using context into a fully initialized one */
30   trait ActionResolver[A <: TurnAction]:

33   /** Generic handler to convert a choice into a validated action */
34   def getActionFromChoice[A <: TurnAction](
35                                             choice: Int,
36                                             context: ActionContext,
37                                             availableActions: List[A]
38                                           )(using resolver: ActionResolver[A]): Either[InputHandlingError, A] =
39     for {
40       action <- availableActions.lift(choice)
41         .toRight(InvalidChoice(choice, availableActions.indices))
42       resolved <- resolver.resolve(action, context)
43     } yield resolved

45   /** Resolver for AiActions */
46   given aiActionResolver: ActionResolver[AiAction] with
47     def resolve(action: AiAction, ctx: ActionContext): Either[InputHandlingError, AiAction] =
48       ctx match
49         case CityContext(cityName, applicableCities) =>
50           action match
51             case _: Sabotage | _: Infect =>
52               Either.cond(
53                 applicableCities.contains(cityName),
54                 action match
55                   case _: Sabotage => Sabotage(List(cityName))
56                   case _: Infect   => Infect(List(cityName))
57                   case _ => action,
58                 InputParsingError(cityName, s"The city '$cityName' is not attackable.")
59               )
60             case Evolve => Right(Evolve)
61         case _ =>
62           Left(InputParsingError("Context", "Invalid context provided for AiAction."))
63 
64   /** Resolver for HumanActions */

82               )

85           Left(InputParsingError("Context", "Invalid context provided for HumanAction."))
